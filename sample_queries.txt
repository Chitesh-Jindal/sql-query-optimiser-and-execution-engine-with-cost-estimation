SELECT EMP.NAME,EMP.AGE FROM EMP WHERE EMP.AGE>25;

SELECT EMP.NAME,EMP.AGE,EMP.SALARY FROM EMP WHERE EMP.AGE>25 AND EMP.SALARY>60000;

SELECT EMP.NAME,EMP.CITY FROM EMP WHERE EMP.CITY='DELHI';

SELECT EMP.NAME,DEPT.NAME,EMP.SALARY FROM EMP JOIN DEPT ON EMP.DEPTID=DEPT.ID WHERE EMP.AGE>25 AND EMP.SALARY>60000;

SELECT EMP.NAME, DEPT.NAME, DEPT.LOCATION, DEPT.BUDGET FROM EMP JOIN DEPT ON EMP.DEPTID = DEPT.ID WHERE DEPT.LOCATION='MUMBAI' AND DEPT.BUDGET>400000;

SELECT CLIENT.NAME,CLIENT.CITY,ASSIGNMENT.CONTRACTVALUE FROM ASSIGNMENT JOIN CLIENT ON ASSIGNMENT.CLIENTID=CLIENT.ID WHERE ASSIGNMENT.CONTRACTVALUE>700000 AND CLIENT.CITY='DELHI';

SELECT EMP.NAME,EMP.SALARY,DEPT.NAME,DEPT.LOCATION FROM EMP JOIN DEPT ON EMP.DEPTID=DEPT.ID WHERE EMP.AGE>25 AND EMP.SALARY>60000 AND DEPT.LOCATION='BANGALORE';
